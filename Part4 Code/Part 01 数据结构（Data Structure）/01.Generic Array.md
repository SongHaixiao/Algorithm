# Generic Array

Target :

1. general types
2. dynamic expansion
3. insert function's time complexity is O(m+n)

### Java

```java
// Java Code

public class GenericArray<T> {
    private T[] data;
    private int count;

    public GenericArray(int capacity){
        this.data = (T[]) new Object[capacity];
        this.count = 0;
    }

    public GenericArray(){this(10);}

    public int getCapacity(){return this.data.length;}

    public int getCount(){return this.count;}

    public boolean isEmpty(){return this.count==0;}

    public void set(int index, T e){
        checkIndex(index);
        this.data[index]=e;
    }

    public T get(int index){
        checkIndex(index);
        return this.data[index];
    }

    public int find(T e){
        for(int i=0;i<this.count;i++){
            if(this.data[i].equals(e)){
                return i;
            }
        }

        return -1;
    }

    public void add(int index, T e){
        checkIndexForAdd(index);

        if(this.count==this.data.length){
            resize(2 * this.data.length);
        }

        for(int i = this.count - 1; i >= index; i--){
            this.data[i+1] = this.data[i];
        }

        this.data[index]=e;
        this.count++;
    }

    public void addHead(T e){add(0,e);}

    public void addTail(T e){add(this.count, e);}

    public T remove(int index){
        checkIndex(index);

        T ret = this.data[index];

        for(int i = index+1; i < this.count;i++){
            this.data[i-1] = this.data[i];
        }

        this.count--;
        this.data[this.count]=null;

        if(this.count==this.data.length/4&&this.data.length/2!=0){
            resize(this.data.length/2);
        }

        return ret;
    }

    public T removeHead(){return remove(0);}

    public T removeTail(){return remove(this.count-1);};

    public void removeElement(T e){
        int index = find(e);
        if(index != -1) {
            remove(index);
        }
    }

    @Override
    public String toString(){
        StringBuilder builder = new StringBuilder();
        builder.append(String.format("Array size = %d, capacity = %d\n", this.count,this.data.length));
        builder.append('[');
        for(int i=0;i<this.count;i++){
            builder.append(this.data[i]);
            if(i!=this.count-1){
                builder.append(", ");
            }
        }

        builder.append(']');
        return builder.toString();
    }

    private void resize(int capacity){
        T[] newData=(T[]) new Object[capacity];

        for(int i=0;i<this.count;i++){
            newData[i]=this.data[i];
        }

        this.data=newData;
    }

    private void checkIndex(int index){
        if(index < 0 || index >= this.count){
            throw new IllegalArgumentException("Add failed! Required index >= 0 and index < size.");
        }
    }

    private void checkIndexForAdd(int index){
        if(index<0||index>this.count){
            throw new IllegalArgumentException("Remove faild! Required");
        }
    }

    public static void main(String[] args){
        GenericArray array = new GenericArray();
        array.addHead(0);
        array.addHead(0);
        array.add(1,5);
        array.add(3,9);
        array.add(3,10);
        array.add(3,11);
        array.addTail(20);
        System.out.println(array.toString());

        array.removeHead();
        array.removeTail();
        array.removeElement(3);
        array.removeElement(11);
        System.out.println(array.toString());

        array.remove(3);
        System.out.println(array.toString());
    }
}
```
